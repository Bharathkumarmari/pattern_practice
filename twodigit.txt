import java.util.*;
class twodigit
{
public static void main(String ar[])
{
Scanner s = new Scanner(System.in);
int n = s.nextInt();
int [] arr=new int[];
for(int i=0;i<n;i++)
{
arr[i]=s.nextInt();
}
for(int i=n-1;i>=0;i--)
{
if(arr[i]>9 && arr[i]<100)
{
System.out.print(arr[i]+" ");
}
else
{
System.out.print("-1");
}
}
